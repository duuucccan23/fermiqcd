///    int nc=3; 
///    int box[]={10,8,8,8};
///    mdp_lattice lattice(4,box);
///    gauge_field U(lattice,nc);
///    mdp_site x(lattice);
///    // set_cold(U);

///       for(int mu=0; mu<U.ndim; mu++)
